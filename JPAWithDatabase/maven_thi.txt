------ ConnectDB ------

public class ConnectDB {
	EntityManagerFactory emf;
	EntityManager em;
	
	public ConnectDB() {
		this.emf = Persistence.createEntityManagerFactory("mssql");
		this.em = emf.createEntityManager();
	}	
	
	public EntityManager getEntityManager() {
		return em;
	}
	
	public void close() {
		em.close();
		emf.close();
	}
}
